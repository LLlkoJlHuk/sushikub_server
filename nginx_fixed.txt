server {
    listen 80;
    server_name 89.111.168.173;

    # React приложение из правильной папки
    root /var/www/sushikub/sushikub_client/dist;
    index index.html;

    # Assets из правильного пути
    location /assets/ {
        alias /var/www/sushikub/sushikub_client/dist/assets/;
        expires 1y;
        add_header Cache-Control "public, immutable";
        try_files $uri =404;
    }

    # API проксирование
    location /api/ {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # ВАЖНО: Изображения С БЕЗ параметрами - отдаем напрямую
    location ~* ^(?!\/assets\/).*\.(webp|jpg|jpeg|png|gif|svg)$ {
        # Проверяем, есть ли параметры запроса
        if ($args = "") {
            # Без параметров - отдаем статику
            alias /var/www/sushikub/sushikub_server/static/;
            expires 1y;
            add_header Cache-Control "public, immutable";
            try_files $uri =404;
        }
        # С параметрами - пропускаем к Node.js серверу
    }

    # Изображения С параметрами - пропускаем к Node.js серверу
    location ~* ^(?!\/assets\/).*\.(webp|jpg|jpeg|png|gif|svg)\? {
        proxy_pass http://localhost:5000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # React SPA
    location / {
        try_files $uri $uri/ /index.html;
    }

    gzip on;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/json;
}
